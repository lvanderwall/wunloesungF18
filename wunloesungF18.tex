	\uH(\jw)&=\frac{\uU_j}{\uV}=\jw L_2+R_2+(\jw L_1+R_1)\|\lr{(}{%
		\jw L_3+R_3\|\frac{1}{\jw C}%
	}{)}\\[2.5mm]%

		(\jw L_1+R_1)\lr{(}{%
			\jw L_3+R_3\|\frac{1}{\jw C}%
		}{)}

		(\jw L_1+R_1)+\lr{(}{%

		}{)}%
	}\ceqn{%
		\cdot\frac{\jw R_3C+1}{\jw R_3C+1}%
	}\\[2.5mm]%

		(\jw L_1+R_1)\lr{(}{%
			(\jw)^2R_3CL_3+\jw L_3+R_3%
		}{)}%
	}{%
		(\jw L_1+R_1)(\jw R_3C+1)+\lr{(}{%

		b_3(\jw)^3+b_2(\jw)^2+b_1(\jw)+b_0%
